# Co-Scientist Agent 상세 기능 설계서

## 1. 데이터 적재 파이프라인

### 1.1 CSV → MariaDB (`load_csv_to_mariadb.py`)

**입력**: `data/sample_paper.csv`
**출력**: MariaDB `paper.sid_v_09_01` 테이블

```
처리 흐름:
1. pandas로 CSV 읽기
2. MariaDB 연결 (SQLAlchemy)
3. 테이블 존재 시 DROP 후 재생성
4. 컬럼 매핑하여 INSERT
5. 인덱스 생성 (filename, doi, coverdate, FULLTEXT(paper_keyword, paper_text))
```

**주의사항**:
- `coverdate`는 INT64 YYYYMMDD 형식으로 저장
- `chunk_id`, `chunk_total_counts`는 CSV에 이미 포함
- NULL 값 처리: 빈 문자열로 대체

### 1.2 MariaDB → Milvus (`load_mariadb_to_milvus.py`)

**입력**: MariaDB `sid_v_09_01`
**출력**: Milvus `m_paper.m_sid_v_09_01` 컬렉션

```
처리 흐름:
1. MariaDB에서 전체 논문 조회
2. 각 논문의 paper_text를 청킹:
   - chunk_size: 512 (CHUNK_SIZE)
   - chunk_overlap: 50 (CHUNK_OVERLAP)
   - 청크마다 chunk_id (1부터), chunk_total_counts 부여
3. 각 청크를 임베딩:
   - EMBEDDING_BASE_URL/v1/embeddings API 호출
   - 모델: bge-m3 (1024차원)
   - 배치 처리 (한 번에 여러 청크)
4. Milvus 컬렉션 생성:
   - Dense 인덱스: IVF_FLAT, metric=IP
   - Sparse: BM25 built-in function (bm25_keywords_sparse)
   - 함수 정의: BM25(paper_text → bm25_keywords_sparse)
5. 배치 INSERT
```

**청킹 로직 (`app/services/chunker.py` → `chunker_v2.py`)**:

현재 `chunker.py`는 고정 크기 문자 분할. 고급 전략은 `chunker_v2.py`에 구현:

```python
    """regex 기반 날짜 파싱 → filters에 저장"""
    query = state["query"]
    filters = state.get("filters") or {}

    # DOI 추출
    doi_match = re.search(r'10\.\d{4,}/\S+', query)
    if doi_match:

        filters["coverdate_to"] = parsed["to"]      # YYYYMMDD int

    state["filters"] = filters
    return state
```

### 4.1 paper_qa (논문 Q&A)

```

[retrieve 노드]
  1. DOI가 filters에 있으면:
     → database.get_paper_fulltext_by_doi(doi) 로 원문 조회
  2. 대화 히스토리에서 논문 제목 추출 가능하면:

     → common.retrieve_by_query(query, filters, top_k=5)

    "당신은 논문 분석 전문가입니다. 제공된 논문 내용을 기반으로 질문에 답하세요.
```

## 5. 공통 헬퍼 함수 상세 (`app/agents/common.py`)

### 5.1 검색 함수

```python
async def retrieve_by_query(
    query: str,
    user_id: str | None,
    filters: dict | None,
    top_k: int = 5
) -> list[dict]:
    """Milvus hybrid search"""
    vector = await embeddings.embed_query(query)
    # filters에서 coverdate_from/to → Milvus expr 변환
    expr = build_filter_expr(filters)  # "coverdate >= 20240101 and coverdate <= 20240630"
    results = await vectorstore.hybrid_search(
        query_vector=vector,
        query_text=query,
        expr=expr,
        top_k=top_k
    )
    return results

async def multi_query_retrieve(
    queries: list[str],
    user_id: str | None,
    filters: dict | None,
    top_k_per_query: int = 3
) -> list[dict]:
    """여러 쿼리로 검색 후 중복 제거"""
    all_results = []
    seen_ids = set()
    for q in queries:
        results = await retrieve_by_query(q, user_id, filters, top_k_per_query)
        for r in results:
            if r["mariadb_id"] not in seen_ids:
                seen_ids.add(r["mariadb_id"])
                all_results.append(r)
    return all_results
```

### 5.2 LLM 호출 함수

```python
async def llm_json_call(
    system_prompt: str,
    user_prompt: str,
    state: AgentState,
    temperature: float = 0.1
) -> dict:
    """LLM 호출 → JSON 파싱"""
    system = inject_date_context(system_prompt, state)
    messages = [
        {"role": "system", "content": system},
        {"role": "user", "content": user_prompt}
    ]
    usage_out = {}
    response = await llm.chat_completion(messages, temperature=temperature, usage_out=usage_out)
    _accumulate_usage(state, usage_out)
    return json.loads(response)  # JSON 파싱

async def llm_text_call(
    system_prompt: str,
    user_prompt: str,
    state: AgentState,
    temperature: float = 0.3
) -> str:
    """LLM 호출 → 텍스트 응답"""
    # stream_mode이면 messages만 저장하고 호출 스킵
    if state.get("metadata", {}).get("_stream_mode"):
        state["metadata"]["_llm_messages"] = [
            {"role": "system", "content": inject_date_context(system_prompt, state)},
            {"role": "user", "content": user_prompt}
        ]
        return ""  # API 레이어에서 스트리밍 처리

    system = inject_date_context(system_prompt, state)
    messages = [
        {"role": "system", "content": system},
        {"role": "user", "content": user_prompt}
    ]
    usage_out = {}
    response = await llm.chat_completion(messages, temperature=temperature, usage_out=usage_out)
    _accumulate_usage(state, usage_out)
    return response
```

### 5.3 날짜 컨텍스트 주입

```python
def inject_date_context(system_prompt: str, state: AgentState) -> str:
    """시스템 프롬프트에 현재 날짜 + 대화 히스토리 주입"""
    date_ctx = get_current_date_context()
    # "현재 날짜: 2026년 3월 25일 (화요일), 2026년 1분기"

    history = state.get("metadata", {}).get("conversation_history", "")

    parts = [system_prompt, f"\n\n[현재 시간 정보]\n{date_ctx}"]
    if history:
        parts.append(f"\n\n[이전 대화]\n{history}")

    return "\n".join(parts)
```

```python
def format_context(search_results: list[dict]) -> str:
    """검색 결과를 LLM 컨텍스트로 변환"""
    parts = []
    for i, r in enumerate(search_results, 1):
        parts.append(f"[논문 {i}]")
        parts.append(f"제목: {r['title']}")
        parts.append(f"저자: {r.get('author', 'N/A')}")
        parts.append(f"DOI: {r.get('doi', 'N/A')}")
        parts.append(f"내용:\n{r['paper_text']}\n")
    return "\n".join(parts)

def build_sources(search_results: list[dict]) -> list[dict]:
    """검색 결과를 SourceDocument 리스트로 변환"""
    return [
        {
            "paper_id": str(r["mariadb_id"]),
            "title": r["title"],
            "author": r.get("author"),
            "doi": r.get("doi"),
            "chunk_id": r.get("chunk_id", 1),
            "chunk_text": r.get("paper_text", "")[:500],
            "score": r.get("score", 0.0),
        }
        for r in search_results
    ]
```

### 5.5 대화 히스토리 참조 해결

```python
def extract_paper_title_from_history(state: AgentState) -> str | None:
    """'이 논문', 'N번째 논문' 등의 참조를 실제 제목으로 변환"""
    query = state["query"]
    sources = state.get("sources", [])

    # "이 논문", "위 논문" → sources[0] 제목
    if re.search(r"(이|위|해당)\s*논문", query) and sources:
        return sources[0]["title"]

    # "N번째 논문" → sources[N-1] 제목
    match = re.search(r"(\d+)\s*번째\s*논문", query)
    if match and sources:
        idx = int(match.group(1)) - 1
        if 0 <= idx < len(sources):
            return sources[idx]["title"]

    # "참조 문헌 N번" → sources의 references에서 추출
    match = re.search(r"참조\s*문헌\s*(\d+)\s*번", query)
    # ...

    return None
```

### 5.6 토큰 사용량 처리

API 응답에는 **최종 답변 생성 LLM 호출의 usage만** 반환한다.
내부 LLM 호출(의도 분류, 조건 추출 등)은 Langfuse에서 개별 추적.

```python
def _set_final_usage(state: dict, usage_out: dict):
    """최종 답변 생성 LLM 호출의 토큰 사용량을 API 응답용으로 설정한다."""
    metadata = state.setdefault("metadata", {})
    metadata["usage"] = {
        "prompt_tokens": usage_out.get("prompt_tokens", 0),
        "completion_tokens": usage_out.get("completion_tokens", 0),
        "total_tokens": usage_out.get("total_tokens", 0),
    }

def _accumulate_usage(state: dict, usage_out: dict):
    """내부 LLM 호출의 토큰 사용량을 _internal_usage에 누적 (디버깅용)."""
    metadata = state.setdefault("metadata", {})
    existing = metadata.setdefault("_internal_usage", {})
    for key in ("prompt_tokens", "completion_tokens", "total_tokens"):
        existing[key] = existing.get(key, 0) + usage_out.get(key, 0)
```

**사용 위치:**
- `llm_json_call` (의도 분류, 조건 추출 등 내부) → `_accumulate_usage` → `_internal_usage`에만 누적
- `llm_text_call` (최종 답변 생성) → `_set_final_usage` → `metadata.usage`에 설정
- `paper_qa`, `analytics`의 직접 LLM 호출 → `_set_final_usage`

**API 레이어:**
- `extract_usage(result)` 헬퍼가 항상 유효한 dict 반환 (값 없으면 0/0/0)
- `ChatResponse.usage`는 필수 필드 (항상 반환, Optional 아님)

```python
class LLMClient:
    def __init__(self):
        self.base_url = settings.LLM_BASE_URL      # http://localhost:20020/v1
        self.api_key = settings.LLM_API_KEY
        self.model = settings.LLM_MODEL
        self.client = httpx.AsyncClient(timeout=120.0)

    @observe(as_type="generation")
    async def chat_completion(
        self,
        messages: list[dict],
        temperature: float = 0.3,
        max_tokens: int = 4096,
        usage_out: dict | None = None
    ) -> str:
        """비스트리밍 LLM 호출"""
        response = await self.client.post(
            f"{self.base_url}/chat/completions",
            json={
                "model": self.model,
                "messages": messages,
                "temperature": temperature,
                "max_tokens": max_tokens,
            },
            headers={"Authorization": f"Bearer {self.api_key}"}
        )
        data = response.json()
        content = data["choices"][0]["message"]["content"]

        # 토큰 사용량 추출
        if usage_out is not None and "usage" in data:
            usage_out.update(data["usage"])

        return content

    async def chat_completion_stream(
        self,
        messages: list[dict],
        temperature: float = 0.3,
        max_tokens: int = 4096,
        usage_out: dict | None = None
    ) -> AsyncGenerator[str, None]:
        """SSE 스트리밍 LLM 호출"""
        async with self.client.stream(
            "POST",
            f"{self.base_url}/chat/completions",
            json={..., "stream": True},
        ) as response:
            async for line in response.aiter_lines():
                if line.startswith("data: "):
                    chunk = json.loads(line[6:])
                    if chunk["choices"][0].get("delta", {}).get("content"):
                        yield chunk["choices"][0]["delta"]["content"]
                    # 마지막 청크에서 usage 추출
                    if "usage" in chunk and usage_out is not None:
                        usage_out.update(chunk["usage"])
```

```python
class VectorStoreClient:
    def __init__(self):
        self.client = MilvusClient(
            uri=f"http://{settings.MILVUS_HOST}:{settings.MILVUS_PORT}",
            db_name=settings.MILVUS_DATABASE
        )

    @observe(name="hybrid_search")
    async def hybrid_search(
        self,
        query_vector: list[float],
        query_text: str,
        expr: str | None = None,
        top_k: int = 5
    ) -> list[dict]:
        """Dense + Sparse Hybrid Search (RRF)"""
        # Dense 검색 요청
        dense_req = AnnSearchRequest(
            data=[query_vector],
            anns_field="embeddings",
            param={"metric_type": "IP", "params": {"nprobe": 10}},
            limit=top_k,
            expr=expr
        )
        # Sparse 검색 요청 (BM25)
        sparse_req = AnnSearchRequest(
            data=[query_text],  # BM25 function이 자동 변환
            anns_field="bm25_keywords_sparse",
            param={"metric_type": "BM25"},
            limit=top_k,
            expr=expr
        )
        # RRF 결합
        results = self.client.hybrid_search(
            collection_name=self.collection,
            reqs=[dense_req, sparse_req],
            ranker=RRFRanker(),
            limit=top_k,
            output_fields=["mariadb_id", "title", "author", "doi", ...]
        )
        return [hit.entity.to_dict() for hit in results[0]]
```

### 7.2 패턴별 regex

```python
# D1: "2024년 11월"
r"(\d{4})년\s*(\d{1,2})월"

# D1: "2024년 10월~12월"
r"(\d{4})년\s*(\d{1,2})월?\s*[~\-]\s*(\d{1,2})월"

# D2: "2023년 N분기"
r"(\d{4})년\s*(\d)분기"

# D2: "2024년 상반기/하반기"
r"(\d{4})년\s*(상반기|하반기)"

# D2: "2022~2024년"
r"(\d{4})\s*[~\-]\s*(\d{4})년?"

# D3: "최근 N개월/년"
r"최근\s*(\d+)\s*(개월|달|년|해)"

# D3: "올해/작년/재작년"
r"(올해|금년|작년|지난해|재작년)"

# D3: 계절
r"(올해|작년|지난)?\s*(봄|여름|가을|겨울)"
# 봄: 3~5월, 여름: 6~8월, 가을: 9~11월, 겨울: 12~2월

# D4: "YYYY년과 YYYY년 비교"
r"(\d{4})년\s*과\s*(\d{4})년\s*비교"
```
